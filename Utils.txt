  sudo docker-compose -f docker-compose.production.yml build --no-cache backend

  # 5. Start all containers
  sudo docker-compose -f docker-compose.production.yml up -d

  # 6. Monitor the backend logs to ensure it starts successfully
  sudo docker-compose -f docker-compose.production.yml logs -f backend

  # 7. Eliminate the old images of containers
  sudo docker system prune -af
